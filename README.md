<h2>How to use the package:</h2>

<code>
<pre>
<strong>
import {
	pixelsToREM,
	remToPixels,
	remToVH,
	vhToREM,
	remToVH,
	vwToREM,
	pixelsToVH,
	pixelsToVW,
	vhToPixels,
	vwToPixels,
} from "web-units-conversify";

console.log(remToPixels(10) + "px");
console.log(pixelsToREM(1024) + "rem");
console.log(remToVH(12) + "vh");
console.log(vwToREM(15) + "em"); // can be EM or REM units
//etc...
</strong>
</pre>
</code>
